+ constructor(props){
+        super(props);
+        this.state = {comments:[]};

+    }

+    getComments(){
+        $.ajax({
+            url:this.props.url,
+            dataType:'json',
+           cache:false,
+            success:data =>{
+                this.setState({comments:data});
+            },

+                console.error(error);
+            }
